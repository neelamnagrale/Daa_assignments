#include <stdio.h>
#include <stdlib.h>
#include <limits.h>
#include <stdbool.h>

#define MAX_NODES 100
#define INF INT_MAX

typedef struct {
    int node;
    int distance;
} MinHeapNode;

typedef struct {
    MinHeapNode *array;
    int *position;
    int size;
} MinHeap;

// Graph structure
typedef struct {
    int vertices;
    int **adjMatrix;  // Adjacency matrix for travel times
} Graph;

// MinHeap functions
MinHeap* createMinHeap(int capacity) {
    MinHeap *heap = (MinHeap*)malloc(sizeof(MinHeap));
    heap->array = (MinHeapNode*)malloc(capacity * sizeof(MinHeapNode));
    heap->position = (int*)malloc(capacity * sizeof(int));
    heap->size = 0;
    return heap;
}

void swap(MinHeapNode *a, MinHeapNode *b) {
    MinHeapNode temp = *a;
    *a = *b;
    *b = temp;
}

void minHeapify(MinHeap *heap, int idx) {
    int smallest = idx;
    int left = 2 * idx + 1;
    int right = 2 * idx + 2;
    
    if (left < heap->size && heap->array[left].distance < heap->array[smallest].distance)
        smallest = left;
    
    if (right < heap->size && heap->array[right].distance < heap->array[smallest].distance)
        smallest = right;
    
    if (smallest != idx) {
        heap->position[heap->array[smallest].node] = idx;
        heap->position[heap->array[idx].node] = smallest;
        swap(&heap->array[smallest], &heap->array[idx]);
        minHeapify(heap, smallest);
    }
}

MinHeapNode extractMin(MinHeap *heap) {
    MinHeapNode root = heap->array[0];
    MinHeapNode lastNode = heap->array[heap->size - 1];
    heap->array[0] = lastNode;
    heap->position[root.node] = heap->size - 1;
    heap->position[lastNode.node] = 0;
    heap->size--;
    minHeapify(heap, 0);
    return root;
}

void decreaseKey(MinHeap *heap, int node, int distance) {
    int i = heap->position[node];
    heap->array[i].distance = distance;
    
    while (i && heap->array[i].distance < heap->array[(i - 1) / 2].distance) {
        heap->position[heap->array[i].node] = (i - 1) / 2;
        heap->position[heap->array[(i - 1) / 2].node] = i;
        swap(&heap->array[i], &heap->array[(i - 1) / 2]);
        i = (i - 1) / 2;
    }
}

bool isInMinHeap(MinHeap *heap, int node) {
    return heap->position[node] < heap->size;
}

// Print path from source to destination
void printPath(int parent[], int j) {
    if (parent[j] == -1) {
        printf("%d", j);
        return;
    }
    printPath(parent, parent[j]);
    printf(" -> %d", j);
}

// Dijkstra's Algorithm
void dijkstra(Graph *graph, int src, int dest) {
    int V = graph->vertices;
    int dist[MAX_NODES];
    int parent[MAX_NODES];
    
    MinHeap *heap = createMinHeap(V);
    
    // Initialize distances and heap
    for (int v = 0; v < V; v++) {
        parent[v] = -1;
        dist[v] = INF;
        heap->array[v].node = v;
        heap->array[v].distance = dist[v];
        heap->position[v] = v;
    }
    
    heap->array[src].distance = dist[src] = 0;
    heap->size = V;
    decreaseKey(heap, src, dist[src]);
    
    printf("\n=== Dijkstra's Algorithm Execution ===\n");
    printf("Finding shortest path from intersection %d to hospital %d\n\n", src, dest);
    
    // Main algorithm loop
    while (heap->size != 0) {
        MinHeapNode minNode = extractMin(heap);
        int u = minNode.node;
        
        // Process all adjacent vertices
        for (int v = 0; v < V; v++) {
            if (graph->adjMatrix[u][v] && isInMinHeap(heap, v) && 
                dist[u] != INF && graph->adjMatrix[u][v] + dist[u] < dist[v]) {
                
                dist[v] = dist[u] + graph->adjMatrix[u][v];
                parent[v] = u;
                decreaseKey(heap, v, dist[v]);
            }
        }
    }
    
    // Print results
    printf("Shortest Distances from Ambulance (Node %d):\n", src);
    printf("--------------------------------------------------\n");
    printf("Intersection\tDistance (min)\tPath\n");
    printf("--------------------------------------------------\n");
    
    for (int i = 0; i < V; i++) {
        printf("%d\t\t", i);
        if (dist[i] == INF)
            printf("INF\t\t-\n");
        else {
            printf("%d\t\t", dist[i]);
            printPath(parent, i);
            printf("\n");
        }
    }
    
    printf("\n=== OPTIMAL ROUTE TO HOSPITAL ===\n");
    printf("Destination: Hospital at Intersection %d\n", dest);
    printf("Total Travel Time: %d minutes\n", dist[dest]);
    printf("Optimal Path: ");
    printPath(parent, dest);
    printf("\n\n");
    
    free(heap->array);
    free(heap->position);
    free(heap);
}

// Create graph
Graph* createGraph(int vertices) {
    Graph *graph = (Graph*)malloc(sizeof(Graph));
    graph->vertices = vertices;
    graph->adjMatrix = (int**)malloc(vertices * sizeof(int*));
    
    for (int i = 0; i < vertices; i++) {
        graph->adjMatrix[i] = (int*)calloc(vertices, sizeof(int));
    }
    
    return graph;
}

// Add edge (road) with travel time
void addEdge(Graph *graph, int src, int dest, int time) {
    graph->adjMatrix[src][dest] = time;
    graph->adjMatrix[dest][src] = time;  // Undirected graph
}

int main() {
    int vertices = 9;  // Number of intersections
    Graph *graph = createGraph(vertices);
    
    printf("=== Smart Traffic Management System ===\n");
    printf("City Road Network with %d intersections\n\n", vertices);
    
    // Sample city road network (travel times in minutes)
    // Adding edges representing roads with traffic conditions
    addEdge(graph, 0, 1, 4);
    addEdge(graph, 0, 7, 8);
    addEdge(graph, 1, 2, 8);
    addEdge(graph, 1, 7, 11);
    addEdge(graph, 2, 3, 7);
    addEdge(graph, 2, 8, 2);
    addEdge(graph, 2, 5, 4);
    addEdge(graph, 3, 4, 9);
    addEdge(graph, 3, 5, 14);
    addEdge(graph, 4, 5, 10);
    addEdge(graph, 5, 6, 2);
    addEdge(graph, 6, 7, 1);
    addEdge(graph, 6, 8, 6);
    addEdge(graph, 7, 8, 7);
    
    int ambulanceLocation = 0;  // Ambulance at intersection 0
    int hospitalLocation = 4;    // Hospital at intersection 4
    
    printf("Ambulance Location: Intersection %d\n", ambulanceLocation);
    printf("Hospital Location: Intersection %d\n", hospitalLocation);
    
    // Run Dijkstra's algorithm
    dijkstra(graph, ambulanceLocation, hospitalLocation);
    
    // Demonstrate dynamic update
    printf("=== DYNAMIC TRAFFIC UPDATE ===\n");
    printf("Traffic congestion detected! Road 2->5 time increased from 4 to 10 minutes\n");
    addEdge(graph, 2, 5, 10);  // Update edge weight
    
    dijkstra(graph, ambulanceLocation, hospitalLocation);
    
    // Cleanup
    for (int i = 0; i < vertices; i++) {
        free(graph->adjMatrix[i]);
    }
    free(graph->adjMatrix);
    free(graph);
    
    return 0;
}
