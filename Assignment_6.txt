#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define MAX_ITEMS 100

typedef struct {
    char name[50];
    int weight;
    int utility;
    int priority;
} ReliefItem;

typedef struct {
    int totalUtility;
    int totalWeight;
    int items[MAX_ITEMS];
    int itemCount;
} KnapsackResult;

int max(int a, int b) {
    return (a > b) ? a : b;
}

KnapsackResult knapsackDP(ReliefItem items[], int n, int capacity) {
    KnapsackResult result = {0, 0, {0}, 0};
    
    int **dp = (int**)malloc((n + 1) * sizeof(int*));
    for (int i = 0; i <= n; i++) {
        dp[i] = (int*)calloc(capacity + 1, sizeof(int));
    }
    
    for (int i = 1; i <= n; i++) {
        for (int w = 0; w <= capacity; w++) {
            if (items[i-1].weight <= w) {
                dp[i][w] = max(dp[i-1][w], 
                              dp[i-1][w - items[i-1].weight] + items[i-1].utility);
            } else {
                dp[i][w] = dp[i-1][w];
            }
        }
    }
    
    result.totalUtility = dp[n][capacity];
    
    int w = capacity;
    for (int i = n; i > 0 && result.totalUtility > 0; i--) {
        if (result.totalUtility != dp[i-1][w]) {
            result.items[result.itemCount++] = i - 1;
            result.totalWeight += items[i-1].weight;
            result.totalUtility = dp[i-1][w - items[i-1].weight] + items[i-1].utility;
            w -= items[i-1].weight;
        }
    }
    
    result.totalUtility = dp[n][capacity];
    
    for (int i = 0; i <= n; i++) {
        free(dp[i]);
    }
    free(dp);
    
    return result;
}

void printResult(ReliefItem items[], KnapsackResult result, const char *method) {
    printf("\n=== %s Results ===\n", method);
    printf("Total Utility Value: %d\n", result.totalUtility);
    printf("Total Weight: %d kg\n", result.totalWeight);
    printf("Items Loaded: %d\n\n", result.itemCount);
    
    printf("%-25s %-10s %-10s %-10s\n", "Item Name", "Weight", "Utility", "Priority");
    printf("----------------------------------------------------------------\n");
    
    for (int i = 0; i < result.itemCount; i++) {
        int idx = result.items[i];
        const char *priorityStr;
        switch(items[idx].priority) {
            case 1: priorityStr = "Critical"; break;
            case 2: priorityStr = "High"; break;
            default: priorityStr = "Normal";
        }
        printf("%-25s %-10d %-10d %-10s\n",
               items[idx].name, items[idx].weight, 
               items[idx].utility, priorityStr);
    }
    printf("----------------------------------------------------------------\n");
}

int main() {
    printf("=== DISASTER RELIEF RESOURCE ALLOCATION SYSTEM ===\n");
    printf("0/1 Knapsack Problem - Earthquake Relief Operation\n\n");
    
    ReliefItem items[] = {
        {"Emergency Medicine", 5, 100, 1},
        {"IV Fluids", 8, 90, 1},
        {"First Aid Kits", 3, 70, 1},
        {"Antibiotics", 4, 95, 1},
        {"Food Packets", 10, 60, 2},
        {"Drinking Water", 15, 75, 2},
        {"Blankets", 7, 50, 3},
        {"Tents", 20, 85, 3},
        {"Flashlights", 2, 40, 3},
        {"Batteries", 3, 45, 3},
        {"Canned Food", 12, 65, 2},
        {"Medical Gloves", 1, 30, 1}
    };
    
    int n = sizeof(items) / sizeof(items[0]);
    int truckCapacity = 50;
    
    printf("Available Relief Items: %d\n", n);
    printf("Truck Capacity: %d kg\n", truckCapacity);
    
    printf("\n=== DYNAMIC PROGRAMMING (OPTIMAL) ===\n");
    
    clock_t start = clock();
    KnapsackResult dpResult = knapsackDP(items, n, truckCapacity);
    clock_t end = clock();
    
    printResult(items, dpResult, "Dynamic Programming");
    
    double timeTaken = ((double)(end - start)) / CLOCKS_PER_SEC * 1000;
    printf("\nExecution Time: %.6f ms\n", timeTaken);
    printf("Time Complexity: O(n * W) = O(%d * %d) = O(%d)\n", 
           n, truckCapacity, n * truckCapacity);
    
    return 0;
}
